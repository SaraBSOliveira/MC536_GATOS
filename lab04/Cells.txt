1) SELECT Food_Code, Ingredient_Num, FCID_Code, Commodity_Weight FROM Recipes 
    WHERE Recipes.Food_Code = 27111300 AND Recipes.Mod_Code = 0;
    
2) SELECT Recipes.Food_Code, Recipes.Ingredient_Num, Recipes.FCID_Code, FCID_Description.FCID_Desc, Recipes.Commodity_Weight FROM Recipes, FCID_Description
    WHERE Recipes.Food_Code = 27111300 AND Recipes.Mod_Code = 0 AND Recipes.FCID_Code = FCID_Description.FCID_Code;
    
3) SELECT FCID_Description.CGN, Crop_Group.Crop_Group_Description FROM Recipes, FCID_Description, Crop_Group
    WHERE Recipes.Food_Code = 27111300 AND Recipes.Mod_Code = 0 AND Recipes.FCID_Code = FCID_Description.FCID_Code AND FCID_Description.CGN = Crop_Group.CGN  AND Crop_Group.CGN = Crop_Group.CGL
    GROUP BY FCID_Description.CGN;
    
4) SELECT FCID_Description.FCID_Desc, COUNT(*) AS Contador
    FROM Recipes, FCID_Description
    WHERE Recipes.FCID_Code = FCID_Description.FCID_Code 
    GROUP BY FCID_Description.FCID_Desc
    ORDER BY Contador DESC;
    
5) SELECT Crop_Group.Crop_Group_Description, AVG(Intake) FROM FCID_Description, Crop_Group, Intake
    WHERE FCID_Description.FCID_Code = Intake.FCID_Code AND FCID_Description.CGN = Crop_Group.CGN  AND Crop_Group.CGN = Crop_Group.CGL
    GROUP BY FCID_Description.CGN;
